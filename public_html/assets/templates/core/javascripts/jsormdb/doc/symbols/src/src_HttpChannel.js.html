1 /**
  2  * @author Avi Deitcher
  3  * @fileOverview Channel to read/write from an HTTP server
  4  */
  5 /*global JSORM */
  6 
  7 /**
  8  * Create new HTTP channel
  9  * @class Channel to communicate over http via Ajax with the back-end server.
 10  * 
 11  * @param {Object} config Configuration object, must have at least one url parameter
 12  * @param {String} config.url URL to use for loading and updating. If it begins with '/', then absolute, else relative.
 13  * @param {String} config.loadUrl URL to use for loading. If it begins with '/', then absolute, else relative.
 14  * @param {String} config.updateUrl URL to use for updating. If it begins with '/', then absolute, else relative.
 15  */
 16 JSORM.db.channel.http = JSORM.extend({}, function(config){
 17 	config = config || {};
 18 	// convenience
 19 	var ajax = JSORM.ajax, fork = JSORM.fork, that = this;
 20 	
 21 	// our URLs; if only one url is given, use it for both
 22 	var loadUrl = config.loadUrl || config.url, updateUrl = config.updateUrl || config.url;
 23 
 24 	// create event-handling
 25 	JSORM.eventualize(this);
 26 	this.events('beforeupdate','update','updateexception','beforeload','load','loadexception');
 27 
 28 	var processResponse = function(eSuccess, eFailure, filename, xhr, success, o) {
 29 		var e, a, s, ct, ct2, res;
 30 		if (success) {
 31 			e = eSuccess; a = o.options; s = true;
 32 			// because both types are sometimes used
 33 			ct = xhr.getResponseHeader("Content-type");
 34 			ct2 = xhr.getResponseHeader("Content-Type");
 35 			res = ct === "text/xml" || ct2 === "text/xml" ? xhr.responseXML : xhr.responseText;
 36 		} else {
 37 			e = eFailure; a = xhr; s = false;
 38 		}
 39 		that.fire(e,o);
 40 		o.callback.call(o.scope,{options: o.arg, success: s, response: res});		
 41 	};
 42 	
 43 	var updateResponse = function(filename, xhr, success, o){
 44 		processResponse("update","updateexception",filename,xhr,success,o);
 45 	};
 46 
 47 	var loadResponse = function(filename, xhr, success, o){
 48 		processResponse("load","loadexception",filename,xhr,success,o);
 49 	};
 50 	
 51 	var message = function(beforeevent, arg, callback, method, url) {
 52 		var params = arg.params, cb = arg.callback, scope = arg.scope, options = arg.options;
 53         if(that.fire("beforeevent", params) !== false){
 54             var  o = {
 55                 params : params || {},
 56                 options: {
 57                     callback : cb,
 58                     scope : scope,
 59                     arg : options
 60                 },				
 61                 callback : callback,
 62 				method: method,
 63                 scope: this,
 64 				url: url
 65             };
 66             ajax(o);
 67         }else{
 68 			fork({fn: cb, scope: scope || that, arg: [{options: options, success: false}]});
 69         }		
 70 	};
 71 	
 72 	JSORM.apply(this,/** @lends JSORM.db.channel.http.prototype */{
 73 		/**
 74 		 * Update the remote data source via http. This is presumed to be asynchronous, and thus will
 75 		 * return before the call is complete. Use a callback to capture the result.
 76 		 * 
 77 		 * @param {Object} [config] Configuration information for the update
 78 		 * @param {Object} [config.params] Parameters to add to the update. Each element is given as a parameter name to the HTTP
 79 		 * 			PUT, while the values are expected to be Strings given as the value of HTTP parameter 
 80 		 * @param {Function} [config.callback] Function to be executed when the update is complete, whether success or failure.
 81 		 *   The callback should expect a single argument, an object, with the following elements:
 82 		 *    <ul>
 83 		 *     <li>success: boolean as to whether or not the update succeeded</li>
 84 		 *     <li>options: the options that were passed to update as config.options</li>
 85 		 *    </ul>
 86 		 * @param {Object} [config.scope] Scope within which to execute the callback
 87 		 * @param {Object} [config.options] Options to pass to the callback
 88 		 */
 89 	    update : function(arg){
 90 			message("beforeupdate",arg,updateResponse, "POST", updateUrl);
 91 		},
 92 	
 93 		/**
 94 		 * Load from the remote data source via http. This is presumed to be asynchronous, and thus will
 95 		 * return before the call is complete. Use a callback to capture the result.
 96 		 * 
 97 		 * @param {Object} [config] Configuration information for the load
 98 		 * @param {Object} [config.params] Parameters to add to the load. Each element is given as a parameter name to the HTTP
 99 		 * 			GET, while the values are expected to be Strings given as the value of HTTP parameter 
100 		 * @param {Function} [config.callback] Function to be executed when the load is complete, whether success or failure.
101 		 *   The callback should expect a single argument, an object, with the following elements:
102 		 *    <ul>
103 		 *     <li>success: boolean as to whether or not the load succeeded</li>
104 		 *     <li>options: the options that were passed to load as config.options</li>
105 		 *    </ul>
106 		 * @param {Object} [config.scope] Scope within which to execute the callback
107 		 * @param {Object} [config.options] Options to pass to the callback
108 		 */
109 		load : function(arg) {
110 			message("beforeload",arg,loadResponse, "GET", loadUrl);
111 		}
112 
113 	});
114 
115 });
