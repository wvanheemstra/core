1 /**
  2  * @author Avi Deitcher
  3  * @fileOverview JSON Parser to convert JSON into objects suitable for jsormdb and vice-versa
  4  */
  5 /*global JSORM */
  6 /**
  7  * Create a new JSON parser
  8  * @class Parser to convert JSON into objects when loaded from a channel and vice-versa
  9  * 
 10  * @param {Object} [config] Configuration parameters
 11  * @param {String} [config.id] Default field to use as the unique identifier field in parsed data
 12  * @param {String} [config.root] Default element to use as the root of actual records in parsed data
 13  */
 14 JSORM.db.parser.json = JSORM.extend({}, function(config){
 15 	config = config || {};
 16 	var id = config.id, root = config.root, lastMeta = {}, lastRoot = {};
 17 	
 18 	// read - input JSON, write out objects
 19 	JSORM.apply(this, /** @lends JSORM.db.parser.json.prototype */{
 20 		/**
 21 		 * Convert JSON into an object structure suitable to load into jsormdb
 22 		 * 
 23 		 * @param {String} json JavaScript Object Notation string with the appropriate information
 24 		 * @returns {Object} An object with the appropriate elements
 25 		 */
 26 		read : function(json) {
 27 			// first parse the data
 28 			var data = null, p;
 29 			p = JSON.parse(json);
 30 
 31 			// data better be a valid object
 32 			if (p && typeof(p) === "object") {
 33 				data = {};
 34 				// if it is an array, just use it directly 
 35 				if (p.isArray) {
 36 					data.records = p;
 37 					data.id = id;
 38 				} else {
 39 					// find out our root and our id
 40 					root = p.meta && p.meta.root ? p.meta.root : root;
 41 					data.records = p[root];
 42 					data.id = p.meta && p.meta.id ? p.meta.id : id;
 43 
 44 					// keep the root information
 45 					lastMeta = p.meta;
 46 					lastRoot = root;
 47 				}
 48 			}
 49 
 50 			return(data);
 51 		},
 52 
 53 		/**
 54 		 * Convert an array of jsormdb objects into JSON as per the original load structure
 55 		 * 
 56 		 * @param {Object[]} records Array of records from a jsormdb database
 57 		 * @returns {String} JSON-encoded String, including appropriate metadata and root
 58 		 */
 59 		write : function(records) {
 60 			// hold our new structure
 61 			var obj = {};
 62 			obj[lastRoot] = records;
 63 			if (lastMeta) {
 64 				obj.meta = lastMeta;
 65 			}
 66 			var j = JSON.stringify(obj);
 67 			if (!j) {throw{message: "JsonParser.write: unable to encode records into Json"};}
 68 			return(j);
 69 		}		
 70 	});
 71 });
