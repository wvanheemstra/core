 60 	var findInternal = function(args) {
 61 		var ret = null, i, len, query, idx, data;
 62 		
 63 		// query the store
 64 		idx = store.query(args.where);
 65 
 66 		// now our idx, if valid, has all of the index entries;
 67 		//  either return the indexes or get all of the actual entries
 68 		if (idx) {
 69 			if (args.index) {ret = idx;}
 70 			else {
 71 				ret = [];
 72 				for (i=0,len=idx.length; i<len;i++) {
 73 					data = store.get(idx[i]);
 74 					ret.push(apply({},clone(data),args.fields));
 75 				}				
 76 			}
 77 		}
 78 
 79 		return(ret);
 80 	};

 92 	var loadData = function(data) {
 93 		var r = data.records;
 94 		clearInternal(false);
 95 		idField = data.id || "id";
 96 		store.addIndex(idField);		
 97 		store.addIndex('type');
 98 		store.insert(r);
 99 		// clean out the journal

103 	/**
104 	 * Load new records - called asynchronously with callback handler.
105 	 * Loading records always means the old transaction is wiped clean and a new transaction is begun
106 	 * immediately *after* the load is complete. In other words, the load itself is not in the journal. 
107 	 * The process is as follows:
108 	 * 1) Wipe journal clean (remove old transaction)
109 	 * 2) Do the load
110 	 * 3) Start a new transaction journal
111 	 */
112 	var loadCallback = function(args){
113 		var options = args.options || {}, r = [], parsed, processed = false;
114 		var e, sfcb, cb = args.callback, scope = args.scope || this;
115 
116 		// only clear and load if we successfully get and parse the data
117 		if (args.success && (parsed = parser.read(args.response))) {
118 			// attempt to load the new data
119 			loadData(parsed);
120 
121 			// we successfully processed
122 	        r = parsed.records;
123 			processed = true;
124 
125 			// specific events
126 			sfcb = options.success;
127 			e = "load";
128 		} else {
129 			sfcb = options.failure;
130 			e = "loadexception";
131 		}
132 
133 		// general event
134         this.fire(e, {records: r, options: options});
135 
136 		// specific success/failure callback
137 		if (sfcb && typeof(sfcb) === "function") {sfcb.call(scope, r, options, processed);}
138 
139 		// always call our callback, whether successful or not
140         if(cb && typeof(cb) === "function"){
141 			cb.call(scope, r, options, processed);
142 		}
143 	};

145     var removeAt = function(index){
146 		var i,len, removed = [], entry;
147 		index = [].concat(index);
148 		for (i=0,len=index.length;i<len;i++) {
149 			entry = store.remove(index[i]);
150 			removed.push(entry);
151 		}
152 		return(removed);
153     };

155 	var write = function(mode) {
156 		var data, tmp, i, len, j, lenj, recs = {}, entry, den, curId, condensed, orig;
157 
158 		// replace mode just dumps it all
159 		if (mode === myclass.modes.replace) {
160 			// get the actual data in the records, no indexes, we don't care about the journal
161 			data = store.get();
162 		} else {
163 			data = [];
164 			/*
165 			 * If we already have an entry, so just update that one
166 			 * What we do with condensed largely depends on what we want to do next and what we did previously
167 			 * - If we are updating an already updated record, we can just add updates
168 			 * - If we are updating a record that was added in this transaction, just change the add data
169 			 * - If we are removing a record, make that the only activity
170 			 * - If we are adding a record, do it straight out
171 			 */
172 			condensed = mode === myclass.modes.condensed;
173 
174 			// get the actual data in the record
175 			for (i=0, len=journal.length; i<len; i++) {
176 				// recall the structure of each journal entry
177 				//  {type: change/add/remove/clear/load, }
178 				entry = journal[i];
179 				if (entry !== null) {
180 					// go through each one, then see what we do
181 					switch(entry.type) {
182 						case myclass.types.change: 
183 							// keep a list of the affected IDs, and make sure they also point to this idx entry
184 							orig = entry.data.original;
185 							for (j=0,lenj=orig.length; j<lenj;j++) {
186 								curId = orig[j].id;
187 								// are we condensed, and did we have a previous entry?
188 								if (condensed && recs[curId]) {
189 									// previous ones will be add or change, so modify that one
190 									// if it was a change, we need to remove it
191 									apply(recs[curId].data,entry.data.changed);
192 								} else {
193 									// either a previous change to this record does not exist, or 
194 									den = {
195 										type: entry.type,
196 										data: clone(entry.data.changed)
197 									};
198 									// save the ID field
199 									den.data[idField] = curId;
200 									recs[curId] = den;
201 									data.push(den);									
202 								}
203 							}
204 							break;
205 						case myclass.types.add:
206 							// the data entry to be sent to the server - we need to get the actual data
207 							tmp = store.get(entry.data);
208 							// store these; there is no way we have a previous journal entry for a newly added record 
209 							for (j=0,lenj=tmp.length;j<lenj;j++) {
210 								den = {
211 									type: entry.type,
212 									data: tmp[j]
213 								};
214 								recs[tmp[j][idField]] = den;
215 								data.push(den);
216 							}
217 							break;
218 						case myclass.types.clear:
219 							data.push({
220 								type: entry.type
221 							});
222 							break;
223 						case myclass.types.remove:
224 							tmp = [];
225 							den = {};
226 							// keep a list of the affected IDs, and make sure they also point to this idx entry
227 							for (j=0, lenj=entry.data.length; j<lenj; j++) {
228 								curId = entry.data[j][idField];
229 								tmp.push(curId);
230 								// are we in condensed mode and there was a previous record?
231 								if (condensed && recs[curId]) {
232 									// previous ones will be add or change, so remove it from the list
233 									recs[curId].data.remove(curId);
234 									// if it was a change, we need to remove it
235 									if (recs[curId].type === myclass.types.change) {
236 										recs[curId] = den;																			
237 									}
238 								} else {
239 									recs[curId] = den;									
240 								}
241 							}
242 							den.type = entry.type; den.data = tmp;
243 							data.push(den);
244 							break;
245 						default:
246 							break;
247 					}					
248 				}
249 			}
250 		}
251 		return(data);
252 	};

254 	/**
255 	 * Handle the results of a write. 
256 	 * 
257 	 * @arg o Object options passed to the original write call
258 	 * @success boolean whether the write succeeded or not
259 	 * @response String full contents of response from the server
260 	 */	
261 	var writeCallback = function(args) {
262 		// if the POST worked, i.e. we reached the server and found the processing URL,
263 		// which handled the processing and responded, AND the processing itself succeeded,
264 		// then success, else exception
265 		var i, len, response = args.response, o = args.options || {}, update;
266 		var r = [], e, sfcb, cb = o.callback, scope = o.scope || this, options = o.options;
267 		var newRec, where, index;
268 
269 		// the expectation for success is that the application itself will determine it
270 		//  via a 'write' handler
271 		if (args.success) {
272 			if (this.fire('write',{options: o, data: response}) !== false) {
273 				// update fields or even whole new records from the server
274 				//  if requested either via options.update or this.updateMode
275 
276 				// we have a few possibilities:
277 				// 1) We replace all our data with that from the server - either we are in mode.replace or we explicitly 
278 				//    said to do so for this write
279 				// 2) We update our data with that from the server, i.e. apply journal changes
280 				// 3) We make no changes to our local data
281 				// which update mode will we work in? Try to use local option, then db-wide, then system default
282 				update = first(o.update,updateMode,defaultUpdateMode);
283 
284 				switch(update) {
285 					case myclass.updates.nothing:
286 						// do nothing
287 						break;
288 					case myclass.updates.replace:
289 						// replace: read our data and then replace everything
290 						r = parser.read(response);
291 						loadData(r);
292 						break;
293 					case myclass.updates.update:
294 						// update: read our data and then go through each record one by one:
295 						// - if a record with this record's ID exists, update
296 						// - if one does not, add it a new
297 						r = parser.read(response);
298 						where = {field: idField, compare: 'equals'};
299 
300 						// we worked in journal mode, so take the changes they recommend and apply them
301 						for (i=0, len=r.records.length; i<len;i++) {
302 							newRec = r.records[i];
303 
304 							// do we have a record with this id? It should be indexed, because it is the ID field
305 							// if it exists, get the original record and update it, else mark it to add
306 							where.value = newRec[idField];
307 							index = findInternal({where: where, index: true});
308 							if (index && index.length > 0) {
309 								store.update(index,newRec);								
310 							} else {
311 								store.insert(newRec);
312 							}
313 						}
314 						break;
315 				}
316 
317 				// clean out the journal
318 				journal.clear();
319 
320 				// EVENTS AND CALLBACKS FOR SUCCESS
321 				// 1) Specific to this transaction
322 				sfcb = o.success;
323 				// 2) All commit registered handlers
324 				e = "commit";
325 			} else {
326 				// some callback said not to complete the write
327 				sfcb = o.failure;
328 				e = "commitexception";
329 			}
330 		} else {
331 			// EVENTS AND CALLBACKS FOR FAILURE
332 			// 1) Specific to this transaction
333 			sfcb = o.failure;
334 			// 2) All write failure registered handlers
335 			e = "writeexception";
336 		}
337 
338 		// general event
339 		this.fire(e,{options: o, data: response});
340 		// success/failure callback
341 		if (sfcb && typeof(sfcb) === "function") {sfcb.call(scope,this,options,response);}
342 		// general callback
343 		if (cb && typeof(cb) === "function") {cb.call(scope,this,options,response);}
344 	};

355 		/**
356 		 * Insert new data directly into the database. The parser will parse.
357 		 * An insert is considered part of a transaction and is logged in the journal. 
358 		 * If you wish to start afresh, use load() instead.

360 		 * @param {Object[]} data Array of data objects to insert into the database
361 		 */
362 		insert : function(data) {
363 			var index;
364 
365 			// parse the data if relevant
366 			if (data) {
367 				// if it is a string, send it to a parser, else use directly
368 				if (typeof(data) === "string") {
369 					data = parser.read(data);
370 					if (data && typeof(data) === "object") {data = data.records;}
371 				}
372 
373 				// use internal function for the insert and log it to the journal
374 				index = store.insert(data);
375 				journal.push({type: myclass.types.add, data: index});
376 
377 				// tell everyone we have added
378 		        this.fire("add", {records: data});
379 			}

384 		 * Search by query. Returns an array of indexes. No matches will return an empty array; invalid query will return null.
385 		 * 
386 		 * @param {Object} params Search parameters
387 		 * @param {Object} params.where Proper query term, either composite or primitive
388 		 * @param {Object} [params.fields] Fields to return. This is an object literal. All fields that are set to non-null and 
389 		 *   have a match will return those fields. Returns all fields if null.
390 		 * @returns {Object[]} Array of the matched records

393 			params = params || {};
394 			var data = findInternal({where: params.where, fields: params.fields, index: false});

398 		/**
399 		 * Update records based on a where clause
400 		 * 
401 		 * @param {Object} params Update parameters
402 		 * @param {Object} params.where Proper query term, either composite or primitive, to determine which records to update.
403 		 *    If null, update all.
404 		 * @param {Object} data New data to enter into all the updated reocrds that match the search term. Single object literal.
405 		 */
406 		update : function(params) {
407 			var index, oldData, det = [], i, len, args = params || {}, id, idconf;
408 			// first find the indexes of all the entries that match the where clause
409 			index = findInternal({where: args.where, index: true});
410 
411 			// get the oldData and update the records
412 			oldData = store.update(index,args.data);
413 			
414 			// get the IDs of those fields
415 			idconf = {};
416 			idconf[idField] = true;
417 			id = store.get(index,idconf);
418 
419 			// create the journal of the change
420 			for (i=0,len=index.length;i<len;i++) {
421 				det.push({index: index[i], data: oldData[i], id: id[i][idField]});
422 			}
423 
424 			// journal the change
425 			journal.push({type: myclass.types.change, data: {changed: args.data, original: det}});
426 
427 			// fire the event
428 			this.fire("update",{records: store.get(index)});
429 		},

432 		/**
433 		 * Load new data to reinitialize this database. This is different from {@link insert} in several ways:
434 		 * <ol>
435 		 * <li>The load (and possible replace) are not considered part of the current transaction. The 
436 		 *     current transaction is terminated, and a new transaction is started immediately after the load</li>
437 		 * <li>The load can come from either data passed directly or from the channel</li>
438 		 * <li>load is always asynchronous, whereas insert is synchronous</li>
439 		 * </ol>
440 		 * 
441 		 * @param {Object} args Arguments to the load.
442 		 * @param {Object} [args.data] Raw data to load. If null, will use the defined channel and parser.
443 		 * @param {Function} [args.callback] Function to call when the load is complete
444 		 * @param {Function} [args.success] Function to call when the load succeeds
445 		 * @param {Function} [args.failure] Function to call when the load fails
446 		 * @param {Object} [args.scope] Scope within which to call the callbacks. 
447 		 * @param {Object} [args.options] Object with options to pass to the callback. 
448 		 */
449 		load : function(args) {
450 			args = args || {};
451 			var params, tp = {callback: args.callback, success: args.success, failure: args.failure, 
452 				scope: args.scope, options: args.options};
453 			// need to insert full load from channel function, followed by loadCallback as an async callback
454 			if (args.data) {
455 				tp.success = true;
456 				tp.response = args.data;
457 				fork({fn: loadCallback, arg: [tp], scope: this});
458 			} else if (channel) {
459 				// load asynchronously via the channel, with loadCallback as the callback
460 
461 				// combine the user params for this call - first the base loadParams, then the per-call params
462 
463 				// if updateParams have been set for this store, set them
464 				params = apply({}, loadParams);
465 				// if particular params have been set for this call, set them
466 				apply(params, args.params);
467 
468 				// add any options
469 				channel.load({params: params, scope: this, callback: loadCallback, options: tp});
470 			} else {
471 				// if no channel was defined, and we were not passed data, we cannot load
472 				tp.success = false;
473 				fork({fn: loadCallback, arg: [tp], scope: this});				
474 			}
475 			return(this);
476 
477 		},

522 		/**
523 		 * Commit the current transaction. If there is a channel, and a non-nothing update mode,
524 		 * it will write to the store. If there is no channel, it will just commit. 
525 		 * The commit mode is determined by the following:
526 		 * <ul>
527 		 * <li>options.mode - for this transaction</li>
528 		 * <li>writeMode - default for this db instance</li>
529 		 * <li>defaultWriteMode - default for all instances of the db</li>
530 		 * </ul>
531 		 * 
532 		 * @param {Object} [options] Commit options
533 		 * @param {Object} [options.mode] Which mode to use for committing, one of the static modes
534 		 * @param {Object} [options.params] Parameters to pass to the channel as part of the update
535 		 * @param {Function} [options.callback] Function to call when the commit is complete
536 		 * @param {Function} [options.success] Function to call when the commit has succeeded
537 		 * @param {Function} [options.failure] Function to call when the commit has failed
538 		 * @param {options.Object} [options.scope] Scope within which to execute the callbacks
539 		 * @param {options.Object} [options.options] Options to pass to the callbacks
540 		 */
541 	    commit : function(options){
542 			options = options || {};
543 			var params, records, mode;
544 			// which mode will we work in? Try to use local option, then store-wide, then global default
545 			mode = first(options.mode,writeMode,defaultWriteMode);
546 
547 
548 			// if there is not channel, we just commit internally
549 			if (!channel || (mode === myclass.modes.nothing)) {
550 				journal.clear();
551 				this.fire("commit",{options: options});			
552 			} else {
553 				if (this.fire("beforewrite",{options: options}) !== false) {
554 					// get the appropriate records - watch out for bad records
555 					records = write(mode);
556 
557 					// combine the user params for this call - first the base updateParams, then 
558 					//   the per-call params. Finally, our privileged params and we can send
559 
560 					// if updateParams have been set for this store, set them
561 					params = apply({}, updateParams);
562 					// if particular params have been set for this call, set them
563 					apply(params, options.params);
564 					// finally, add all our params
565 					apply(params,{
566 						data: parser.write(records),
567 						mode: mode
568 					});
569 
570 					// add any options
571 		            channel.update({params: params, callback: writeCallback, scope: this, options: options});			
572 				}
573 			}

576 		/**
577 		 * Reject a transaction. If given a count, it will reject the last count activities. If given no count,
578 		 * a count of 0, or a count greater than the total number of activities in this transaction, it will
579 		 * reject the entire transaction.
580 		 * 
581 		 * @param {Integer} count Number of steps within the transaction to reject. If empty, 0, or greater than the 
582 		 *   total number of steps, the entire transaction will be rejected.
583 		 */
584 	    reject : function(count){
585 			// are we rejecting all or some?
586 			var start = 0, index, data, type, i, j, len, lenj, orig;
587 			if (!count || count > journal.length) {
588 				count = journal.length;
589 				start = 0;
590 			} else {
591 				start = journal.length - count;
592 			}
593 
594 			// back out the last 'count' changes in reverse order
595 			// get the last 'count' elements of the journal
596 			var m = journal.splice(start,count).reverse();
597 			for (i=0, len=m.length; i<len; i++) {
