  8 /** 
  9  * @namespace Container for all engine components, and parent for included engines
 10  */
 11 JSORM.db.engine = function(){
 12 	var apply = JSORM.apply, clone = JSORM.clone;
 13 	var compares, pass1, pass2, pass3, intersection, union, keysAsArray, isPrimitive, isCompound;
 14 	
 15 	compares = {
 16 		equals: function(name,val) {return(function(entry){return(entry[name]===val);});},
 17 		"in": function(name,val) {
 18 				var h, ret;
 19 				if (val.isArray) {h=val.hasher(); ret = function(entry){return(h.hasOwnProperty(entry[name]));};}
 20 				else {ret = null;}
 21 				return(ret);
 22 			},
 23 		gt: function(name,val) {
 24 				return(typeof(val) === "number" ? function(entry){return(entry[name]>val);} : null);
 25 			},
 26 		ge: function(name,val) {
 27 				return(typeof(val) === "number" ? function(entry){return(entry[name]>=val);} : null);
 28 			},
 29 		lt: function(name,val) {
 30 				return(typeof(val) === "number" ? function(entry){return(entry[name]<val);} : null);
 31 			},
 32 		le: function(name,val) {
 33 				return(typeof(val) === "number" ? function(entry){return(entry[name]<=val);} : null);
 34 			},
 35 		starts: function(name,val) {
 36 				return(typeof(val) === "string" ? function(entry){return(entry[name].indexOf(val) === 0);} : null);
 37 			},
 38 		ends: function(name,val) {
 39 				return(typeof(val) === "string" ? function(entry){var a = entry[name]; return(a.length-a.indexOf(val)-val.length === 0);} : null);
 40 			},
 41 		contains: function(name,val) {
 42 				return(typeof(val) === "string" ? function(entry){return(entry[name].indexOf(val) >= 0);} : null);
 43 			},
 44 		isnull: function(name,val) {return(function(entry){return(entry[name]===null);});},
 45 		notnull: function(name,val) {return(function(entry){return(entry[name]!==null);});}
 46 	};	
 47 
 48 	intersection = function() {
 49 		var result,i,len,o;
 50 		if (!arguments || arguments.length<1) {
 51 			result = {};
 52 		} else if (arguments.length == 1 && typeof(arguments[0]) === "object") {
 53 			result = arguments[0];
 54 		} else {
 55 			result = arguments[0].isArray ? arguments[0].hasher() : arguments[0];
 56 			for (i=1,len=arguments.length;i<len;i++) {
 57 				o = arguments[i].isArray ? arguments[i].hasher() : arguments[i];
 58 				result = JSORM.common(result,o,true);
 59 			}
 60 		}
 61 		return(result);
 62 	};
 63 	
 64 	union = function() {
 65 		var result,i,len,o;
 66 		if (!arguments || arguments.length<1) {
 67 			result = {};
 68 		} else {
 69 			result = {};
 70 			for (i=0,len=arguments.length;i<len;i++) {
 71 				o = arguments[i].isArray ? arguments[i].hasher() : arguments[i];
 72 				result = JSORM.apply(result,o);
 73 			}					
 74 		}
 75 		return(result);
 76 	};
 77 	
 78 	keysAsArray = function(o) {
 79 		var i, r = [];
 80 		for (i in o) {
 81 			if (i && o.hasOwnProperty(i) && typeof(o[i]) !== "function") {r.push(i);}
 82 		}		
 83 		return(r);
 84 	};
 85 	
 86 	isPrimitive = function(where) {
 87 		return(where.hasOwnProperty('field') && where.field && typeof(where.field) === "string" &&
 88 			where.hasOwnProperty('compare') && where.compare && compares[where.compare] && 
 89 			(where.hasOwnProperty("value") || where.compare === "isnull" || where.compare === "notnull"));
 90 	};
 91 	isCompound = function(where) {
 92 		return(where.hasOwnProperty("join") && (where.join === "and" || where.join === "or") && 
 93 				where.hasOwnProperty("terms") && where.terms.isArray);
 94 	};
 95 	
 96 	/**
 97 	 * First pass against the query tree. Attempts to match any primitive against the index.
 98 	 * 
 99 	 * @param {Object} where A standard query term, either composite or primitive
100 	 * @param {Object} index The index
101 	 * @return {Object} Results tree, where each primitive is either a function to pass a record or an array of result indexes
102 	 * @private
103 	 */
104 	pass1 = function(where,index) {
105 		var r, r2, i, len, subm;
106 
107 		// is it a primitive?
108 		if (isPrimitive(where)) {
109 				// can we get a result from the index?
110 				if ((subm = index.find(where)) !== null) {
111 					r = subm;
112 				} else {
113 					// we cannot get from index, so create the function that will process any values
114 					r = compares[where.compare](where.field,where.value);
115 				}
116 		} else if (isCompound(where)) {
117 			// is it a compound?
118 			r = {join: where.join, terms:[], fn:[],comps:[]}
119 			if (where.type) {r.type = where.type};
120 			for (i=0, len=where.terms.length; i<len; i++) {
121 				r2 = pass1(where.terms[i],index);
122 				// determine if it is a list of indexes, or a function
123 				if (r2.isArray) {
124 					// indexes, so we merge appropriately
125 					// if it is and, we want the union; if not, the intersection
126 					r.terms.push(where.terms[i]);
127 				} else if (typeof(r2) === "function"){
128 					// function, so we keep each one
129 					r.fn.push(r2);
130 				} else {
131 					// another compound
132 					r.comps.push(r2);					
133 				}
134 			}
135 		} else {
136 			r = null;
137 		}
138 		return(r);						
139 	};
140 	
141 	/**
142 	 * Second pass against the query tree. Resolve any functions using the resultant intersections or all records
143 	 * 
144 	 * @param {Object} where A query tree, output of pass1(), where each primitive is a fn() or []
145 	 * @param {}
146 	 * @return {Object} Results tree, where each primitive is a set of indexes
147 	 * @private
148 	 */
149 	pass2 = function(where,foreach,index,limit) {
150 		// q is a function that returns null, unless it explicitly becomes valid
151 		var r = [], r2, r3, subquery, i, len, j, lenj, list, keeper, typelimit;
152 		
153 		/*
154 		 * How does this work? Everything passed will be like a compound.
155 		 * 		A: join AND: 
156 		 * 				1) take the intersection of any earlier terms
157 		 * 				2) take those results, and feed each one into each function. Those for which every function returns 
158 		 * 					true, we keep; others are discarded
159 		 * 				3) take those results, and use them as a limit. Feed those as the limiting factor into 
160 		 * 					each sub-compound
161 		 * 			Any that survive all three steps are valid.
162 		 * 		B: join OR:
163 		 * 				1) take the union of any earlier terms
164 		 * 				2) take the limit, or the entire data set, and feed each one into each function. Those for which any
165 		 * 					function returns true, we keep; others are discarded
166 		 * 				3) take the limit, or the entire data set, and feed each one into each sub-compound. Union the results
167 		 * 					of each compound into the total set.
168 		 * 			Any that survive any one step are valid.
169 		 */
170 		// is there a type limit?
171 		if (where.type) {
172 			typelimit = index.find({field:'type',compare:'equals',value:where.type});
173 			limit = limit ? intersection(limit,typelimit) : typelimit.hasher();
174 		}
175 		
176 		if (where.join === "and") {
177 			// AND join - intersection
178 			
179 			// was there any limit to start?
180 			if (limit) {r2 = limit;}
181 			
182 			// 1) if we had any where terms, further restrict
183 			if (where.terms && where.terms.length>0) {
184 				// first merge all of the previous terms
185 				r3 = intersection.apply(this,where.terms);
186 				r2 = r2 ? intersection(r3,r2) : r3;
187 			}
188 			
189 			// 2) feed the matching function into foreach - keep only those that match every function
190 			if (where.fn && where.fn.length > 0) {
191 				// go through each one from before
192 				r2 = foreach(function(record) {
193 					// will we keep this?
194 					keeper = true;
195 					for (i=0,len=where.fn.length;i<len;i++) {
196 						if (!where.fn[i](record)) {
197 							// it did not match even one function, and we are doing intersection AND,
198 							//  so skip entirely
199 							keeper = false;
200 							break;
201 						}
202 					}
203 					return(keeper);
204 				},r2?keysAsArray(r2):null);
205 			}
206 			
207 			// intersection with any sub-compounds - must be limited to r2
208 			if (where.comps && where.comps.length>0) {
209 				for (i=0,len=where.comps.length;i<len;i++) {
210 					// AND = intersection, therefore only those in both the sub-compound *and* 
211 					//    the current r2 are kept.
212 					r2 = pass2(where.comps[i],foreach,index,r2);
213 				}
214 			}
215 
216 		} else {
217 			// OR join - union
218 			
219 			// take the limit (if any), else the entire data set as our starting point
220 			// feed that into the first function
221 			// the results of the first function are saved
222 			// feed the limit (if any), else the entire data set into the second function 
223 			// add those results to the results of the first function
224 			// repeat for all of the functions
225 			// results are all are the final set
226 
227 			// 1) use the limit or entire data set as a starting point
228 
229 			// union with the previous terms from the indexed output
230 			if (where.terms.length>0) {
231 				r2 = union.apply(this,where.terms);
232 				r2 = limit ? intersection(r2,limit) : r2;
233 			}
234 			
235 			
236 			// 2) feed the function into foreach, adding the results to the final set
237 			if (where.fn.length > 0) {
238 				r3 = foreach(function(record){
239 					var matched = false;
240 					for (i=0,len=where.fn.length;i<len;i++) {
241 						// go through each function; as soon as one is matched on this entry, keep it and go to next
242 						//      index entry
243 						if (where.fn[i](record)) {
244 							matched = true; 
245 							break;
246 						}
247 					}
248 					return(matched);
249 				},limit);
250 				r2 = r2 ? union(r2,r3) : r3.hasher();					
251 			}
252 			
253 			// 3) results are all of the final set, the union of the output of all functions, i.e. r2
254 			
255 			// union with any sub-compounds
256 			if (where.comps.length>0) {
257 				for (i=0,len=where.comps.length;i<len;i++) {
258 					if ((r3 = pass2(where.comps[i],foreach,index,limit)) && r3.isArray) {
259 						for (j=0,lenj=r3.length; j<lenj;j++) {
260 							r2[r3[j]] = true;
261 						}
262 					}
263 				}
264 			}
265 		}
266 		// r2 now contains a hash, where each key is a valid index, and each value is true;
267 		//  just turn it into an array in r
268 		if (r2) {
269 			r = r2.isArray ? r2 : keysAsArray(r2);				
270 		} else {
271 			r = [];
272 		}
273 
274 		// we have now devolved an entire compound of primitives into a single array of indexes
275 		//     which is precisely what we wanted
276 		return(r);
277 	};
278 	
279 
280 	return /** @lends JSORM.db.engine  */{
281 		/**
282 		 * Construct a query function from a where statement that is suitable to testing each record in a table for 
283 		 *   a full-table scan
284 		 * 
285 		 * @param {Object} where Standard search term, either primitive or composite
286 		 * @returns {Function} A function that takes a single javascript object, i.e. a table record, 
287 		 *    as an argument and reports if it matches by returning a boolean: true or false
288 		 */
289 		executeQuery : function(where,index,foreach) {
290 			var i, len, subm, match = [], idx, fn, results, tmp;
291 
292 			// if the where is blank, just return them all
293 			if (!where) {
294 				results = foreach(function(record){
295 					return(true);
296 				});
297 			} else {
298 				// before we do it, the root of our tree must always be a compound
299 				if (isPrimitive(where)) {
300 					tmp = {join:'and'};
301 					if (where.type) {tmp.type = where.type;}
302 					delete where.type;
303 					tmp.terms = [where];
304 					where = tmp;
305 				}
306 				
307 				// two passes
308 				// first pass: for each primitive, convert to results from index or function
309 				//             for each compound, split terms of the compound into: compounds, results or functions
310 				results = pass1(where,index);
311 
312 				// second pass: go through the tree, resolve each function by passing it the results of the intersection (AND)
313 				//    or all of the records (OR) and mergin the results together
314 				//  
315 				results = pass2(results,foreach,index);				
316 			}
317 			
318 			
319 			return(results);		
320 		}		
321 	};
322 	
323 }();

325 /** 
326  * Create new JSORM.db.engine.array.
327  * @class Array-based in-memory storage engine.<br/>
328  * Note: array engine does not support indexing
329  */
330 JSORM.db.engine.array = JSORM.extend(JSORM.db.engine,function() {
331 	this.type = "array";
332 	var data = [], index = null;
333 	var apply = JSORM.apply;
334 	// the looping function
335 	var foreach = function(fn,limit) {
336 		var i, len, r = [];
337 		if (limit && limit.isArray)  {
338 			for (i=0,len=limit.length;i<len;i++) {
339 				if (fn(data[limit[i]])) {
340 					r.push(limit[i]);
341 				}
342 			}
343 		} else {
344 			for (i=0,len=data.length;i<len;i++) {
345 				if (fn(data[i])) {
346 					r.push(i);
347 				}
348 			}
349 		}
350 		return(r);			
351 	};

397 		/**
398 		 * Get records at one or more locations. 
399 		 * Equivalent of "select * where index in [index]" or "select * where index = index"
400 		 * 
401 		 * @param {Integer|Integer[]} index A location or array of locations whose records are desired.
402 		 * @param {Object} fields An object indicating which fields of the records at index to retrieve. The object should
403 		 *    have one element with a value of true for those elements in the record desired in the results. If the fields
404 		 *    argument is null or undefined, all fields are returned.
405 		 * @returns {Object|Object[]} The fields desired for the records selected, either a single record if index is an
406 		 *    integer or an array of record of index is an array.
407 		 */
408 		get : function(idx, fields) {
409 			var ret, i, len;
410 			if (idx === null || typeof(idx) === "undefined") {
411 				ret = data;
412 			} else if (idx && idx.isArray) {
413 				ret = [];
414 				for (i=0, len=idx.length; i<len; i++) {
415 					ret.push(apply({},data[idx[i]],fields));
416 				}
417 			} else {
418 				ret = apply({},data[idx],fields);
419 			}
420 			return(ret);
421 		},
422 
423 		/**
424 		 * Update records at one or more locations. 
425 		 * Equivalent of "update newData where index in [index]" or "update newData where index = index"
426 		 * 
427 		 * @param {Object} newData An object with the data to replace at the desired indexes. 
428 		 * @param {Integer|Integer[]} index A location or array of locations whose records are desired to be updated.
429 		 * @returns {Object[]} The changed fields of the old records.
430 		 */
431 		update : function(idx, newData) {
432 			var r, i, len, oldData = [], changes;
433 			for (i=0,len=idx.length; i<len; i++) {
434 				// get the existing record
435 				r = data[idx[i]];
436 				if (r) {
437 					// keep the old data
438 					changes = {};
439 					// for each entry in the new data, keep the old data at that entry, and then overwrite it in the
440 					//   core data store
441 					apply(changes,r,newData);
442 					apply(r,newData);
443 					oldData[i] = changes;
444 				}			
445 			}
446 
447 			// return the old data for the journalling
448 			return(oldData);
449 		},

476 		 * @return {Integer[]} Array of indexes that match the query
477 		 */
478 		query : function(where) {
479 			return(this.executeQuery(where,index,foreach));
480 		}

487 /** 
488  * Create new JSORM.db.engine.hash
489  * @class Hash-based in-memory storage engine
490  * 
491  * @param {Object} index A pre-constructed index to use for this table storage engine. If none is passed, use the default
492  *    JSORM.db.index.hash.
493  */
494 JSORM.db.engine.hash = JSORM.extend(JSORM.db.engine,function(index) {
495 	this.type = "hash";
496 	var data = {}, length = 0, max = 0, unused = [];
497 	var apply = JSORM.apply; 
498 	index = index || JSORM.db.index.hash();
499 
500 	var foreach = function(fn,limit) {
501 		var i,len,r = [];
502 		if (limit) {
503 			for (i=0,len=limit.length;i<len;i++) {
504 				if (fn(data[limit[i]])) {
505 					r.push(limit[i]);
506 				}
507 			}
508 		} else {
509 			for (i in data) {
510 				if (fn(data[i])) {
511 					r.push(i);
512 				}
513 			}
514 		}
515 		return(r);
516 	};
517 
518 	apply(this,/** lends JSORM.db.engine.hash.prototype */{
519 		/**
520 		 * Determine how many records are in the database. Equivalent of "select count(index)"
521 		 * 
522 		 * @returns {Integer} Number of records
523 		 */
524 		length : function() {
525 			return(length);
526 		},
527 
528 		/**
529 		 * Insert an arbitrary number of records into the database. 
530 		 * 
531 		 * @param {Object|Object[]} records The records to insert, either a single JavaScript object or an array of objects.
532 		 */
533 		insert : function(records) {
534 			var i, len, idx, locs = [];
535 
536 			// all records are added at the next available slot
537 			for (i=0,len=records.length;i<len;i++) {
538 				// the place we put it is either at the next unused spot, or at the max, which must then be incremented
539 				if (typeof(idx = unused.shift()) === "undefined") {idx = max++;}
540 				data[idx] = records[i];
541 				// save the index where it was
542 				locs.push(idx);
543 
544 				// extend the length
545 				length++;
546 			}
547 
548 			// add the new records to the index
549 			index.add(locs,records);
550 
551 			return(locs);
552 		},
553 
554 		/**
555 		 * Remove a single record from the database at a particular location. 
556 		 * 
557 		 * @param {Object} index The internal index location at which to remove the record
558 		 * @returns {Object} The removed record
559 		 */
560 		remove : function(idx) {
561 			var entry = data[idx];
562 			delete data[idx];
563 			// remove from the index, if relevant
564 			index.remove(idx,entry);
565 			length--;
566 			// reduce the max, if we just reduced the last one
567 			if (idx+1 === max) {
568 				max--;
569 			} else {
570 				unused.push(idx);
571 			}
572 			return(entry);
573 		},
574 
575 		/**
576 		 * Clear all records from the database.
577 		 */
578 		clear : function() {
579 			// clear out the data
580 			JSORM.clear(data);
581 			// clear out the index
582 			index.clear();
583 			// clear out the unused
584 			unused.clear();
585 			// mark that we are empty
586 			length = 0;
587 			// the max is 0
588 			max = 0;
589 		},
590 
591 		/**
592 		 * Get records at one or more locations. 
593 		 * Equivalent of "select * where index in [index]" or "select * where index = index"
594 		 * 
595 		 * @param {Integer|Integer[]} index A location or array of locations whose records are desired.
596 		 * @param {Object} fields An object indicating which fields of the records at index to retrieve. The object should
597 		 *    have one element with a value of true for those elements in the record desired in the results. If the fields
598 		 *    argument is null or undefined, all fields are returned.
599 		 * @returns {Object|Object[]} The fields desired for the records selected, either a single record if index is an
600 		 *    integer or an array of record of index is an array.
601 		 */
602 		get : function(idx,fields) {
603 			var ret, i, len;
604 			if (idx === null || typeof(idx) === "undefined") {
605 				// need to return as an array
606 				ret = [];
607 				for (i in data) {
608 					if (i && typeof(i) !== "function" && typeof(data[i]) === "object") {ret.push(data[i]);}
609 				}
610 			} else if (idx && idx.isArray) {
611 				ret = [];
612 				for (i=0, len=idx.length; i<len; i++) {
613 					ret.push(apply({},data[idx[i]],fields));
614 				}
615 			} else {
616 				ret = apply({},data[idx],fields);
617 			}
618 			return(ret);
619 		},
620 
621 		/**
622 		 * Update records at one or more locations. 
623 		 * Equivalent of "update newData where index in [index]" or "update newData where index = index"
624 		 * 
625 		 * @param {Object} newData An object with the data to replace at the desired indexes. 
626 		 * @param {Integer|Integer[]} index A location or array of locations whose records are desired to be updated.
627 		 * @returns {Object[]} The changed fields of the old records.
628 		 */
629 		update : function(idx, newdata) {
630 			var r, i, len, oldData = [], changes;
631 			idx = [].concat(idx);
632 			for (i=0,len=idx.length; i<len; i++) {
633 				// get the existing record
634 				r = data[idx[i]];
635 				if (r) {
636 					// keep the old data
637 					changes = {};
638 					// for each entry in the new data, keep the old data at that entry, and then overwrite it in the
639 					//   core data store
640 					apply(changes,r,newdata);
641 					apply(r,newdata);
642 					oldData[i] = changes;
643 					// update the index
644 					index.update(changes,newdata,idx[i]);				
645 				}			
646 			}
647 
648 			// return the old data for the journalling
649 			return(oldData);
650 		},
