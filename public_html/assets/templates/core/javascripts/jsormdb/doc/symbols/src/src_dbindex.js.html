1 /**
  2  * @author adeitcher
  3  * @fileOverview Indexes for jsormdb. Currently provides only in-memory hash
  4  */
  5 /*global JSORM */
  6 
  7 /** 
  8 * Create new JSORM.db.index.hash
  9  * @class Hash-based in-memory index for a database table. Supports only equals matches. For < <= > >= starts, use a B-tree.
 10  * 
 11  * @param {String|String[]} fields Single name of field or array of field names to initially index. Can be changed later.
 12  */
 13 JSORM.db.index.hash = JSORM.extend({},function(f) {
 14 	this.type = "hash";
 15 	var fields = 0, data = {};
 16 	
 17 	JSORM.apply(this,/** @lends JSORM.db.index.hash.prototype */{
 18 		/**
 19 		 * Add fields to the index. If the field is already indexed, do nothing.
 20 		 * 
 21 		 * @param {String|String[]} f Name or array of names of fields to index
 22 		 */
 23 		fields : function(f) {
 24 			var i, len;
 25 			if (f) {
 26 				f = [].concat(f);
 27 				for (i=0,len=f.length; i<len; i++) {
 28 					// only need to register it if it is a string and we do not yet have it
 29 					if (typeof(f[i]) === "string" && !data.hasOwnProperty(f[i])) {
 30 						data[f[i]] = {};
 31 						fields++;
 32 					}
 33 				}
 34 			}
 35 		},
 36 
 37 		/**
 38 		 * Remove fields from the index. If the field is not indexed, do nothing.
 39 		 * 
 40 		 * @param {String|String[]} f Name or array of names of fields to remove from the index
 41 		 */
 42 		unfields : function(f) {
 43 			var i, len;
 44 			if (f) {
 45 				f = [].concat(f);
 46 				for (i=0,len=f.length; i<len; i++) {
 47 					// only need to unregister it if it is a string and we already have it
 48 					if (typeof(f[i]) === "string" && data.hasOwnProperty(f[i])) {
 49 						delete data[f[i]];
 50 						fields--;
 51 					}
 52 				}
 53 			}
 54 		},
 55 
 56 		/**
 57 		 * Add one or more records to the index, including the location where they are located. The location
 58 		 * is expected to be internal to the table engine implementation and have no meaning outside of that engine.
 59 		 * 
 60 		 * @param {Array} index Array of internal location reference pointers for the added records
 61 		 * @param {Object[]} records Full set of records to index. This array must be precisely the same length as the index
 62 		 *   array.
 63 		 */
 64 		add : function(index, records) {
 65 			var i,j,len, ci, dj, rij;
 66 			// add to the index only if something has been indexed
 67 			if (fields > 0) {
 68 				// work as an array
 69 				records = [].concat(records);
 70 				index = [].concat(index);
 71 				// go through each indexed field, for each record
 72 				for (i=0, len=records.length; i<len; i++) {
 73 					ci = index[i];
 74 					for (j in data) {
 75 						// if this is a property in the index data, and it exists on the record, record it
 76 						if (data.hasOwnProperty(j) && records[i].hasOwnProperty(j)) {
 77 							dj = data[j]; rij = records[i][j];
 78 							dj[rij] = dj[rij] || [];
 79 							dj[rij].push(ci);
 80 						}
 81 					}
 82 				}
 83 			}
 84 
 85 		},
 86 
 87 		/**
 88 		 * Remove single record from the index, either by index or by record. First preference is record, if blank then index.
 89 		 * 
 90 		 * @param {Object} index Indexed location, internal to the storage engine
 91 		 * @param {Object} record Actual record to remove
 92 		 */
 93 		remove : function(index, record) {
 94 			var j;
 95 			// first try by record
 96 			for (j in data) {
 97 				if (data.hasOwnProperty(j) && record.hasOwnProperty(j)) {
 98 					// remove the reference to this index
 99 					data[j][record[j]].remove(index);
100 				}
101 			}		
102 		},
103 
104 		/**
105 		 * Clear the index
106 		 */
107 		clear : function() {
108 			var i;
109 			for (i in data) {
110 				if (data.hasOwnProperty(i)) {
111 					JSORM.clear(data[i]);
112 				}
113 			}
114 		},
115 
116 		/**
117 		 * Find all records that fit a particular query. There are three possible responses:
118 		 * <ul>
119 		 * <li>match - found some records that match the query, hence will return an array of locations</li>
120 		 * <li>nomatch - able to perform the query, but found no matches, hence return an empty array</li>
121 		 * <li>noquery - unable to perform the query because one or more of the following is true:
122 		 *   <ul>
123 		 * 		<li>The query type compares clause is not indexable by this index, e.g. "contains"</li>
124 		 * 		<li>The query type is invalid or not a primitive</li>
125 		 * 		<li>The query type field is not indexed</li>
126 		 * </ul>
127 		 * For example, a match returns an array [1,5,789]; nomatch returns an empty array []; noquery returns null. 
128 		 * 
129 		 * @param {Object} query A standard search term; a primitive will be accepted while a composite will be ignored
130 		 * @returns {Array} Array of matches locations, internal to the storage engine, empty if not matches, null if field
131 		 *    is not indexed or the search type is not compatible 
132 		 */
133 		find : function(query) {
134 			var ret = null, field;
135 			// first check if this is something we can match
136 			if (query && query.field && query.compare && (field = data[query.field]) && query.compare === "equals") {
137 				// we return the indexes where it matches
138 				ret = field[query.value];
139 			}
140 			return(ret);
141 		},
142 
143 		/**
144 		 * Update the information in one record. If a field is changed, and that field is indexed mark it as changed.
145 		 * 
146 		 * @param {Object} old The old data for the record. Only the changed data should be passed.
147 		 * @param {Object} newdata The new data for the record. Only the changed data should be passed.
148 		 * @param {Object} index The index of the record.
149 		 */
150 		update : function(old,newdata,index) {
151 			var i, field;
152 			// check each field if it is indexed
153 			for (i in newdata) {
154 				if (newdata.hasOwnProperty(i) && data.hasOwnProperty(i) && (field = data[i]) && old[i] != newdata[i]) {
155 					// if the field is indexed, change the value for a particular index. Here it is indexed,
156 					//  so we remove the index from the old value and add it to the new
157 					field[old[i]].remove(index);
158 					field[newdata[i]].push(index);
159 				}
160 			}
161 		}
162 	});
163 	// initialize fields
164 	this.fields(f);
165 	
166 });
